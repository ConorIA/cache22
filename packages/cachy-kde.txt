# ─── Image format plugins (HEIC/AVIF/JXL in Dolphin/Gwenview) ──
# kimageformats adds Qt image-plugin support for HEIF, AVIF, JXL, KRA,
# ORA, PSD, EXR, HDR, etc. — gives Dolphin thumbnails and Gwenview
# previews for everything modern phones produce. qt6-imageformats
# covers the rest (TIFF, JPEG2000, MNG, ICNS, WBMP, …). The libheif/
# libavif/libjxl deps come along automatically; listed here only as
# defensive pins in case the plugin metapackage drifts.
kimageformats
qt6-imageformats
libheif
libavif
libjxl
libraw                   # camera raw (CR2/CR3/NEF/ARW/…) thumbnails
webp-pixbuf-loader       # GTK app .webp support (GIMP, Eye of GNOME, …)

# ─── GStreamer codec stack (system-wide audio/video decode) ────

# VA-API path) needs to decode the formats people actually have.

gst-plugins-good
